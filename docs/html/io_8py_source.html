74  @staticmethod
75  def _get_dynamic_pet_volume(sorted_files: SortedDict):
76 
77  """Reads a 4D dynamic PET volume as an Image from a SortedDict with each frame's Frame Reference Time
78  (0054, 1300) attribute as keys and a list of corresponding DICOM files as values, sorted with regard to their
79  respective Image Position (Patient) (0020, 0032) z-coordinate.
80 
81  Parameters
82  ----------
83  sorted_files : SortedDict
84  A SortedDict with, for each frame, its Frame Reference Time (0054, 1300) attribute as key and as value, a
85  list of the corresponding files sorted with regard to their respective Image Position (Patient) (0020, 0032)
86  z-coordinate.
87 
88  Returns
89  -------
90  dynamic_volume : Image
91  The 4D dynamic PET volume in output units.
92  time_array : ndarray
93  A 1D array specifying the mid-frame time in seconds for each frame of dynamic_volume wrt the injection time.
94 
95  """
96 
97  series_reader = sitk.ImageSeriesReader()
98  series_reader.SetGlobalWarningDisplay(False)
99  series_reader.SetLoadPrivateTags(True)
100  series_reader.SetMetaDataDictionaryArrayUpdate(True)
101 
102  reference_files = list(sorted_files.values())[0]
103  series_reader.SetFileNames(reference_files)
104  reference_image = series_reader.Execute()
105 
106  series_date = series_reader.GetMetaData(0, '0008|0021')
107  series_time = series_reader.GetMetaData(0, '0008|0031')
108  series_datetime = series_date + series_time
109 
110  # SimpleITK cannot read DICOM SQ elements
111  # TODO: Remove extra dependency to pydicom
112  meta_data = pydicom.dcmread(reference_files[0], stop_before_pixels=True)
113 
114  if [0x0054, 0x0016] in meta_data:
115  radiopharmaceutical_sequence = meta_data[0x0054, 0x0016][0]
116  if [0x0018, 0x1078] in radiopharmaceutical_sequence:
117  injection_datetime = radiopharmaceutical_sequence[0x0018, 0x1078].value
118  elif [0x0018, 0x1072] in radiopharmaceutical_sequence:
119  injection_datetime = series_date + radiopharmaceutical_sequence[0x0018, 0x1072].value
120  else:
121  injection_datetime = series_datetime
122  warnings.warn('Injection time could not be retrieved. Series time will be used.')
123  else:
124  injection_datetime = series_datetime
125  warnings.warn('Injection time could not be retrieved. Series time will be used.')
126 
127  if series_reader.HasMetaDataKey(0, '0008|0070'):
128  manufacturer = series_reader.GetMetaData(0, '0008|0070').lower()
129  else:
130  manufacturer = 'unknown'
131  warnings.warn('Unknown manufacturer. Frame Reference Time will be considered relative to mid-frame.')
132 
133  series_time = datetime.strptime(series_datetime, '%Y%m%d%H%M%S.%f')
134  injection_time = datetime.strptime(injection_datetime, '%Y%m%d%H%M%S.%f')
135  injection_delay = (injection_time - series_time).total_seconds() * 1000.0
136 
137  dimensions = reference_image.GetSize()
138  number_of_frames = len(sorted_files)
139 
140  dynamic_pet_array = np.zeros((dimensions[2], dimensions[1], dimensions[0], number_of_frames), dtype=np.float64)
141  time_array = np.zeros(number_of_frames, dtype=np.float64)
142  index = 0
143 
144  for reference_time, files in sorted_files.items():
145 
146  series_reader.SetFileNames(files)
147  image = series_reader.Execute()
148 
149  mid_frame_time = reference_time - injection_delay
150  # GE's Frame Reference Time (0054, 1300) refers to the start of the frame, so the half Actual Frame Duration
151  # (0018, 1242) is added
152  if 'ge' in manufacturer and series_reader.HasMetaDataKey(0, '0018|1242'):
153  mid_frame_time += 0.5*(float(series_reader.GetMetaData(0, '0018|1242')))
154 
155  time_array[index] = mid_frame_time/1000.0
156  dynamic_pet_array[:, :, :, index] = sitk.GetArrayViewFromImage(image)
157  index += 1
158 
159  dynamic_volume = sitk.GetImageFromArray(dynamic_pet_array)
160  dynamic_volume.CopyInformation(reference_image)
161 
162  return dynamic_volume, time_array

164  @staticmethod
165  def _get_sorted_dynamic_pet_dicom_files(directory: str):
166 
167  """Sorts dynamic PET DICOM files from a root directory.
168 
169  Overlapping frames for a given Study Instance UID (0020, 000D) reconstructed in different series will be
170  gathered provided that they share the same Series Date (0008, 0021) and Series Time (0008, 0031) attributes.
171 
172  Note that only images of Modality (0008, 0060) 'PT' and Series Type (0054, 1000) 'DYNAMIC\\IMAGE' will be
173  considered.
174 
175  Parameters
176  ----------
177  directory : str
178  The root directory containing all dynamic PET DICOM files for the series. Will be walked recursively.
179 
180  Returns
181  -------
182  sorted_files : list of SortedDict
183  A list containing, for each Study Instance UID/Series Datetime combination, a SortedDict with each found
184  frame's Frame Reference Time (0054, 1300) attribute as key and as value a list of the corresponding files
185  sorted wrt their respective Image Position (Patient) (0020, 0032) z-coordinate.
186 
187  """
188 
189  file_reader = sitk.ImageFileReader()
190  sorted_dict = {}
191 
192  for root, dirs, files in walk(directory, topdown=False):
193  for file in files:
194  path = join(root, file)
195  file_reader.SetFileName(path)
196  file_reader.ReadImageInformation()
197  try:
198  modality = file_reader.GetMetaData('0008|0060')
199  if 'PT' in modality:
200  series_type = file_reader.GetMetaData('0054|1000')
201  if 'DYNAMIC\\IMAGE' in series_type:
202  study_id = file_reader.GetMetaData('0020|000d')
203  series_datetime = file_reader.GetMetaData('0008|0021') + \
204  file_reader.GetMetaData('0008|0031')
205  reference_time = float(file_reader.GetMetaData('0054|1300'))
206  slice_location = float(file_reader.GetMetaData('0020|0032').split('\\')[-1])
207  if study_id not in sorted_dict:
208  sorted_dict[study_id] = {}
209  if series_datetime not in sorted_dict[study_id]:
210  sorted_dict[study_id][series_datetime] = SortedDict()
211  if reference_time not in sorted_dict[study_id][series_datetime]:
212  sorted_dict[study_id][series_datetime][reference_time] = SortedDict()
213  sorted_dict[study_id][series_datetime][reference_time][slice_location] = path
214  except RuntimeError:
215  warnings.warn(f'Required metadata could not be read for file \'{path}\'. File will be ignored.')
216 
217  sorted_files = []
218 
219  for study_id in sorted_dict.keys():
220  for series_datetime in sorted_dict[study_id].keys():
221  series_dict = SortedDict()
222  for reference_time in sorted_dict[study_id][series_datetime].keys():
223  series_dict[reference_time] = list(sorted_dict[study_id][series_datetime][reference_time].values())
224  sorted_files.append(series_dict)
225 
226  return sorted_files

228  @staticmethod
229  def read(path: str):
230 
231  """Reads dynamic PET DICOM files from a root directory into a single 4D dynamic PET volume.
232 
233  Overlapping frames reconstructed in different series will be merged provided that they share the same Series
234  Date (0008, 0021) and Series Time (0008, 0031) attributes.
235 
236  If more than one dynamic volume is found in the root directory, the first one will be returned.
237 
238  Parameters
239  ----------
240  path : str
241  The root directory containing all dynamic PET DICOM files for the series. Will be walked recursively.
242 
243  Returns
244  -------
245  dynamic_volume : Image
246  The 4D merged dynamic PET volume in output units.
247  time_array : ndarray
248  A 1D array specifying the mid-frame time in seconds for each frame of dynamic_volume w.r.t. the injection
249  time.
250 
251  """
252 
253  assert isdir(path)
254 
255  sorted_files = DICOMDynamicVolumeReader._get_sorted_dynamic_pet_dicom_files(path)
256 
257  if len(sorted_files) == 0:
258  raise RuntimeError('No dynamic PET volume found in the specified directory.')
259 
260  elif len(sorted_files) > 1:
261  warnings.warn('More than one dynamic PET volume found in the specified directory, returning the first one.')
262 
263  dynamic_volume, time_array = DICOMDynamicVolumeReader._get_dynamic_pet_volume(sorted_files[0])
264 
265  return dynamic_volume, time_array
