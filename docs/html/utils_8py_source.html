26 def extract_curves_from_volume(dynamic_volume: SimpleITK.Image, mask: SimpleITK.Image, neighborhood: int=0):
27 
28  """Extracts all curves from a 4D volume within a 3D binary mask as a 2D array.
29 
30  The curves are averaged within a (2*neighborhood+1) cubic neighborhood around the corresponding voxel, truncated to
31  the volume bounds.
32 
33  Parameters
34  ----------
35  dynamic_volume : Image
36  A 4D volume.
37  mask : Image
38  A 3D binary mask. Must have the same spatial dimensions as dynamic_volume.
39  neighborhood: int, optional
40  The half width of the neighborhood for curve averaging. Default: 0.
41 
42  Returns
43  -------
44  curves : ndarray
45  A 2D array containing for all the extracted curves (axis 0) the corresponding samples (axis 1).
46 
47  """
48 
49  dynamic_volume_array = sitk.GetArrayViewFromImage(dynamic_volume)
50  mask_array = sitk.GetArrayViewFromImage(mask)
51 
52  assert dynamic_volume_array.ndim == 4 and mask_array.ndim == 3
53  assert mask_array.shape[:3] == dynamic_volume_array.shape[:3]
54 
55  shape = dynamic_volume_array.shape
56  voxels = np.where(mask_array != 0.0)
57  slices = [tuple([slice(max(v[i]-neighborhood, 0), min(v[i]+neighborhood+1, shape[i])) for i in range(3)])
58  for v in zip(*voxels)]
59  curves = np.array([np.mean(dynamic_volume_array[s], axis=(0, 1, 2)) for s in slices])
60 
61  return curves

154 def make_volume_from_curves(curves, mask, default_value=0.0):
155 
156  """Recreates a 4D volume from an array of curves originating from a 3D binary mask.
157 
158  The curves must be stored in the same order as non-zero voxels of the binary mask are iterated over (x->y->z). This
159  typically corresponds to the ordering of the curves returned by dpatk.utils.extract_curves_from_volume.
160 
161  Parameters
162  ----------
163  curves : ndarray
164  A 2D array containing for all the provided curves (axis 0) the corresponding samples (axis 1).
165  mask : Image
166  A 3D binary mask. Must have the same number of non-zero elements as the number number of provided curves.
167  default_value : float, optional
168  The default sample value for curves outside the binary mask. Default: 0.0.
169 
170  Returns
171  -------
172  dynamic_volume : Image
173  A 4D volume containing the mean frame.
174 
175  """
176 
177  assert curves.ndim == 2
178 
179  mask_array = sitk.GetArrayFromImage(mask)
180  voxels = np.where(mask_array != 0)
181 
182  assert curves.shape[0] == len(voxels[0])
183 
184  dynamic_volume_array = default_value*np.ones(mask_array.shape[:3] + (curves.shape[-1],), dtype=np.float64)
185  dynamic_volume_array[voxels] = curves
186  dynamic_volume = sitk.GetImageFromArray(dynamic_volume_array)
187  dynamic_volume.CopyInformation(mask)
188 
189  return dynamic_volume
