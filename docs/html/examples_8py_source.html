28 def read_write_volume(input_directory, output_file):
29 
30  """
31  Reads a 4D dynamic PET volume sorted as DICOM files possibly belonging to different series with overlapping
32  frames and stores it as a single MHA file.
33 
34  Parameters
35  ----------
36  input_directory : str
37  The root directory containing all dynamic PET DICOM files for the series. Will be walked recursively.
38  output_file : str
39  The path of the output dynamic PET file.
40 
41  """
42 
43  # First, let us load the dynamic volume from the directory as a SimpletITK::Image.
44  # An array containing the mid-frame times wrt to the tracer injection is also returned.
45  dynamic_volume, time_array = DICOMDynamicVolumeReader.read(input_directory)
46 
47  # Then, the sorted dynamic volume is stored as a single MHA file to ease further processing.
48  SimpleVolumeWriter.write(dynamic_volume, output_file)

51 def register_volume(input_file, output_file):
52 
53  """
54  Rigidly registers all frames of a 4D dynamic PET volume by mutual information maximization.
55 
56  More information on the method used can be found in Martens et al. Cancers 13(10):2342. 2021.
57 
58  Parameters
59  ----------
60  input_file : str
61  The path of the dynamic PET file in MHA format.
62  output_file : str
63  The path of the output registered dynamic PET file.
64 
65  """
66 
67  # First, let us load the sorted dynamic volume stored as a single file.
68  dynamic_volume = SimpleVolumeReader.read(input_file)
69 
70  # Then, let us define the frame indices of each block for the registration as a list of ranges.
71  # Default values for 906 overlapping frames of 20" spaced by 2". Must be adjusted for each specific framing.
72  block_indices = [range(i, i+30) for i in range(36, 900, 30)]
73 
74  # Let us now launch the registration. The 4D registered volume is returned as a SimpletITK::Image along with the
75  # original and registered blocks (4D SimpletITK::Image) for quality check and the corresponding registration
76  # parameters (2D numpy::ndarray).
77  # In this example, the delete_original_volume flag is set to true to save RAM usage. This means that dynamic_volume
78  # should no longer be used after this call.
79  registered_dynamic_volume, original_blocks, registered_blocks, parameters = \
80  register_dynamic_volume(dynamic_volume, block_indices, delete_original_volume=True)
81 
82  # Finally, let us save the results as single MHA files.
83  SimpleVolumeWriter.write(registered_dynamic_volume, output_file)
84  SimpleVolumeWriter.write(original_blocks, join(dirname(output_file), 'original_blocks.mha'))
85  SimpleVolumeWriter.write(registered_blocks, join(dirname(output_file), 'registered_blocks.mha'))

88 def run_decomposition_analyses(dynamic_volume_file, mask_file):
89 
90  """
91  Runs the decomposition analyses (FA and PCA) of a 4D dynamic PET volume within a 3D binary mask.
92 
93  Parameters
94  ----------
95  dynamic_volume_file : str
96  The path of the dynamic PET file in MHA format.
97  mask_file : str
98  The path of the binary mask file in MHA format.
99 
100  """
101 
102  # First, let us load the registered dynamic volume and the binary mask stored as single files.
103  dynamic_volume = SimpleVolumeReader.read(dynamic_volume_file)
104  mask = SimpleVolumeReader.read(mask_file)
105 
106  # Then, extract all curves within the mask from the dynamic volume as a 2D numpy::ndarray.
107  # A division of the resulting array by 1000 is performed to avoid overflow for data expressed in BQML.
108  # The neighborhood argument is set to 0, which means that no average smoothing of the curves is performed.
109  data = extract_curves_from_volume(dynamic_volume, mask, neighborhood=0)/1000.0
110 
111  # Define the desired number of components for the decomposition models.
112  nb_of_components = 6
113 
114  # Define the desired decomposition models.
115  models = [FAModel, PCAModel]
116 
117  for Model in models:
118 
119  # Instantiate the model and fit the data.
120  model = Model(nb_of_components=nb_of_components)
121  model.fit(data)
122 
123  # Transform the curves into model parameter (component) values.
124  parameters = model.transform(data)
125 
126  # Rebuild a 4D volume from the parameter (component) values and the binary mask.
127  components_volume = make_volume_from_curves(parameters, mask)
128 
129  # Finally, let us write the resulting volume as a single MHA file.
130  SimpleVolumeWriter.write(components_volume, join(dirname(dynamic_volume_file),
131  f'{model.get_name().lower()}.mha'))

134 def run_all_examples(directory):
135 
136  """
137  Successively runs all the examples from a root DICOM directory.
138 
139  Notes: - A 3D binary mask with the same dimensions as the dynamic PET volume must be drawn and stored in MHA format
140  in the root directory prior to the analysis.
141  - The default frame indices of the blocks used for dynamic volume registration were determined for 906
142  overlapping frames of 20" spaced by 2" and must be adjusted according to the specific framing.
143 
144  Parameters
145  ----------
146  directory : str
147  The root directory containing the dynamic PET DICOM files and the MHA binary mask file.
148 
149  """
150 
151  read_write_volume(directory, join(directory, 'volume.mha'))
152  register_volume(join(directory, 'volume.mha'), join(directory, 'registered_volume.mha'))
153  run_decomposition_analyses(join(directory, 'registered_volume.mha'), join(directory, 'mask.mha'))
