101 class DecompositionModelBase(ModelBase):
102 
103  """
104  Base class for decomposition dynamic PET models derived from the sklearn.decomposition module.
105 
106  Parameters
107  ----------
108  nb_of_components : int
109  The number of components (parameters) of the model. Must be > 0.
110 
111  Attributes
112  ----------
113  _is_fitted : bool
114  Specifies whether the model is fitted.
115  _model : BaseEstimator
116  The sklearn internal decomposition model.
117 
118  """
119 
120  def __init__(self, nb_of_components: Optional[int] = None):
121 
122  super().__init__(nb_of_components)
123 
124  # Forces the class to be abstract without any abstractmethod.
125  if self.__class__ is DecompositionModelBase:
126  raise TypeError(f'Can\'t instantiate abstract class {DecompositionModelBase.__name__}')
127 
128  assert nb_of_components is None or nb_of_components > 0
129 
130  self._is_fitted_is_fitted = False
131  self._model_model = None
132 
133  def fit(self, data):
134 
135  """Fits the model to the dynamic PET data.
136 
137  Parameters
138  ----------
139  data : ndarray
140  A 2D array with cases on axis 0 and dynamic PET samples on axis 1.
141 
142  """
143 
144  assert data.ndim == 2
145  assert data.shape[0] >= data.shape[1]
146  assert data.shape[1] >= self._nb_of_parameters_nb_of_parameters
147 
148  self._model_model.fit(data)
149  self._is_fitted_is_fitted = True
150 
151  def generate(self, parameters):
152 
153  """Generates dynamic PET data from model parameter (component) values.
154 
155  Parameters
156  ----------
157  parameters : ndarray
158  A 2D array with cases on axis 0 and model parameter (component) values on axis 1.
159 
160  Returns
161  -------
162  data : ndarray
163  A 2D array with cases on axis 0 and dynamic PET samples on axis 1.
164 
165  """
166 
167  assert self._is_fitted_is_fitted
168  assert 0 < parameters.ndim < 3
169  assert parameters.shape[-1] <= self._model_model.n_components_
170 
171  if parameters.ndim == 1:
172  parameters = parameters[np.newaxis, ...]
173 
174  components = np.zeros(parameters.shape[0], self._model_model.n_components_)
175  components[:, :parameters.shape[-1]] = parameters
176  data = self._model_model.inverse_transform(components)
177 
178  if data.shape[0] == 1:
179  data = np.squeeze(data, axis=0)
180 
181  return data
182 
183  def get_components(self):
184 
185  """Returns the model components.
186 
187  Returns
188  -------
189  components : ndarray
190  A 2D array with for each component (axis 0), the corresponding weights of the original data features
191  (axis 1).
192 
193  """
194 
195  assert self._is_fitted_is_fitted
196 
197  components = self._model_model.components_[:self._nb_of_parameters_nb_of_parameters]
198 
199  return components
200 
201  def transform(self, data):
202 
203  """Transforms dynamic PET data into model parameter (component) values.
204 
205  Parameters
206  ----------
207  data : ndarray
208  A 2D array with cases on axis 0 and dynamic PET samples on axis 1.
209 
210  Returns
211  -------
212  parameters : ndarray
213  A 2D array with cases on axis 0 and model parameter (component) values on axis 1.
214 
215  """
216 
217  assert self._is_fitted_is_fitted
218  assert 0 < data.ndim < 3
219  assert data.shape[-1] == self._model_model.components_.shape[1]
220 
221  if data.ndim == 1:
222  data = data[np.newaxis, ...]
223 
224  parameters = self._model_model.transform(data)[:, :self._nb_of_parameters_nb_of_parameters]
225 
226  if parameters.shape[0] == 1:
227  parameters = np.squeeze(parameters, axis=0)
228 
229  return parameters

252 
253  """

259 
260  assert tolerance > 0.0
261  assert max_iter > 0
262  assert rotation is None or rotation in ['varimax', 'quartimax']
263 
264  # All component values are computed, then returned values are truncated to _nb_parameters.
265  self._model_model_model = FactorAnalysis(n_components=None, tol=tolerance, copy=True, max_iter=max_iter,

307  Parameters

309  nb_of_components : int
310  The number of components (parameters) of the model. Must be > 0.
311  whiten : bool, optional

314  Specifies whether the incremental algorithm must be used. Set it to True for large datasets. Default: False.

316  The batch size for the incremental algorithm. Ignored if incremental is False. Default: None, set to 5 times the

318

321  _model : PCA
322  The sklearn internal ICA model.
323 
324  """
325 
326  def __init__(self, nb_of_components: int, whiten: bool = False, incremental: bool = True,
327  batch_size: Optional[int] = None):
328 
329  super().__init__(nb_of_components)
330 
331  assert batch_size is None or batch_size > 0
332 
333  # All component values are computed, then returned values are truncated to _nb_parameters.
334  if incremental:
335  self._model_model_model = IncrementalPCA(n_components=None, whiten=whiten, copy=True, batch_size=batch_size)
336  else:
337  self._model_model_model = PCA(n_components=None, copy=True, whiten=whiten, svd_solver='auto', tol=0.0,
338  iterated_power='auto', random_state=None)
339  self._name_name_name = 'PCA'
